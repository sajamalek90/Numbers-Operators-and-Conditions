Numbers, Operators, and Conditions

Number

The numbers can be integers or floating-point numbers. They are stored as 64-bit floating-point numbers. JavaScript numbers all the same.

let num1 = 42; // Integer => number
let num2 = 3.14; // Floating-point number => number
        console.log(1000000); // number
        console.log(1_000_000); // Sentactic sugar
        console.log(1e6); // e
        console.log(1000000.00); // with Decimal
        console.log(BigInt(999999999999)); // Number + BigInt (Es6)
        console.log(Number.MAX_SAFE_INTEGER); // 9007199254740991 => Number Max safe
        console.log(Number.MAX_VALUE); // 1.7976931348623157e+308 => Number Max Value

Number Methods

JavaScript provides several methods to work with numbers, such as toString(), toFixed(), and parseInt().

let num = 123.456;
        console.log(num.toString()); // "123.456"
        console.log(100..toString()); // "100"
        console.log(num.toFixed(2)); // "123.46"

        console.log(parseInt(num)); // 123
        console.log(parseInt("100 Saja")); // 100
        console.log(parseFloat(num)); // 123.456
    
        console.log(Number.isInteger("100")); // false => (Es6)
        console.log(Number.isInteger(100.56)); // false
        console.log(Number.isInteger(100)); // true

        console.log(Number.isNaN("Saja")); // false => (Es6)
        console.log(Number.isNaN("Saja" / 50)); // true

Math Object

The Math object provides mathematical functions and constants, such as Math.PI, Math.sqrt(), and Math.random().

console.log(Math.round(99.2)); // 99
console.log(Math.round(99.6)); // 100
console.log(Math.ceil(99.2)); // 100
console.log(Math.floor(99.9)); // 99

console.log(Math.min(10, 50, -30, 0, 100)); // -30
console.log(Math.max(10, 50, -30, 0, 100)); // 100

console.log(Math.random()); // Random number between 0 and 1

console.log(Math.abs(-99)); // 99
console.log(Math.pow(2, 4)); // 16

console.log(Math.trunc(99.5)); // 99 "Return only integer part" (Es6)

console.log(Math.PI); // 3.141592653589793
console.log(Math.sqrt(16)); // 4

String Methods

JavaScript provides several methods to manipulate strings, such as toUpperCase(), toLowerCase(), and substring().

let str = "Hello, World!";
        console.log(str[1]); // "e"
        console.log(str[13]); // undefined
        
        console.log(str.charAt(1)); // "e"
        console.log(str.charAt(13)); // 

        console.log(str.length); // 13

        str = " Hello, World! ";
        console.log(str.length); // 15
        console.log(str.trim()); // "Hello, World!"
        console.log(str.length); // 13

        str = "Hello, World!";
        console.log(str.toUpperCase()); // "HELLO, WORLD!"
        console.log(str.toLowerCase()); // "hello, world!"

        ==================================================

        // indexOf(Value [Mandatory], Start [Optional] 0)
            console.log(str.indexOf("World")); // 7
            console.log(str.indexOf("World", 8)); // -1
            console.log(str.indexOf("o")); // 4

        // lastIndexOf(Value [Mandatory], Start [Optional] length)
            console.log(str.lastIndexOf("o")); // 8

        // slice(Start [Mandatory], End [Optional] Not include end)
            console.log(str.slice(2)); // "llo, World!"
            console.log(str.slice(2, 5)); // "llo"
            console.log(str.slice(-2)); // "d!"
            console.log(str.slice(-6, -2)); // "worl"
        
        // repeat(Times) => Es6
            console.log(str.repeat(3)); // "Hello, World! Hello, World! Hello, World!"

        // split(Separator [Optional], Limit [Optional])
            console.log(str.split()); // ["Hello World!"]
            console.log(str.split("")); // ["H", "e", "l", "l", "o", " ", "W", "o", "r", "l", "d", "!"]
            console.log(str.split("", 5)); // ["H", "e", "l", "l", "o"]

        ===============================================================================================

        // substring(Start [Mandatory], End [Optional] Not include end)
            console.log(str.substring(0)); // "Hello, World!"
            console.log(str.substring(2, 6)); // "llo,"
            console.log(str.substring(6, 2)); // "llo,"
            console.log(str.substring(-10)); // "Hello, World!"
            console.log(str.substring(-10, 6)); // "Hello,"
            console.log(str.substring(str.length - 1)); // "!"
        
        // substr(Start [Mandatory], Characters to extract)
            console.log(str.substr(0)); // "Hello, World"
            console.log(str.substr(0, 6)); // "Hello,"
            console.log(str.substr(17)); // 
            console.log(str.substr(-3)); // "ld!"
            console.log(str.substr(-6, 2)); // "Wo"
        
        // includes(Value [Mandatory], Start [Optional] 0) => Es6
            console.log(str.includes("World")); // true
            console.log(str.includes("Hello", 6)); // false
        
        // startsWith(Value [Mandatory], Start [Optional] 0) => Es6
            console.log(str.startsWith("H")); // true
            console.log(str.startsWith("H", 2)); // false
            console.log(str.startsWith("l", 2)); // true
        
        // endsWith(Value [Mandatory], Length [Optional] full length) => Es6
            console.log(str.endsWith("!")); // true
            console.log(str.endsWith("!", 2)); // false
            console.log(str.endsWith("l", 3)); // true

Comparison Operators

Comparison operators are used to compare two values. They return a boolean value (true or false).

console.log(10 > 5); // true
console.log(10 == "10"); // true (loose equality)
console.log(10 === "10"); // false (strict equality)

Logical Operators

Logical operators are used to combine multiple conditions. They include && (AND), || (OR), and ! (NOT).

console.log(true && false); // false
console.log(true || false); // true
console.log(!true); // false

If Condition

The if statement is used to execute a block of code if a condition is true.

let age = 18;
        if (age >= 18) {
            console.log("You are an adult.");
        }

Nested If Condition

Nested if statements are used to check multiple conditions.

let age = 18;
        let hasLicense = true;
        if (age >= 18) {
            if (hasLicense) {
                console.log("You can drive.");
            }
        }

Conditional Ternary Operator

The ternary operator is a shorthand for the if-else statement.

let age = 18;
        let status = (age >= 18) ? "Adult" : "Minor";
        console.log(status); // "Adult"

Nullish Coalescing Operator

The nullish coalescing operator (??) returns the right-hand operand if the left-hand operand is null or undefined.

let value = null;
        let result = value ?? "Default Value";
        console.log(result); // "Default Value"

Switch Statement

The switch statement is used to perform different actions based on different conditions.

let day = "Monday";
        switch (day) {
            case "Monday":
                console.log("Start of the workweek");
                break;
            case "Friday":
                console.log("End of the workweek");
                break;
            default:
                console.log("Midweek");
        }
